%% Neural network
\newcommand{\OxfordNeuralNet}{%
  \begin{scope}[x=0.52cm,y=0.43cm,transform shape]
    \readlist\Nnod{4,6,6,5,3}
    \foreachitem \N \in \Nnod{%
      \foreach \i [evaluate={\x=\Ncnt; \y=\N/2-\i+0.5; \prev=int(\Ncnt-1);}] in {1,...,\N}{%
        \node[circle,minimum size=3.2mm,inner sep=0pt,
              draw=mintDark!70,fill=mint!60,line width=.5pt,opacity=.70]
          (ONet\Ncnt-\i) at (\x,\y) {};
        \ifnum\Ncnt>1
          \foreach \j in {1,...,\Nnod[\prev]}{%
            \draw[line width=.22pt,mintDark!60,opacity=.28]
              (ONet\prev-\j) -- (ONet\Ncnt-\i);
          }
        \fi
      }
    }
  \end{scope}%
}

%% Manifold surface
\newcommand{\OxfordSurface}{%
  \tdplotsetmaincoords{63}{130}%
  \begin{scope}[tdplot_main_coords,scale=.78,transform shape]
    \tikzmath{function f(\x) {return 1.15 + 0.22*cos(1.8*\x r) - 0.13*sin(3.2*\x r);};}
    \pgfmathsetmacro{\dominio}{2.25}
    \pgfmathsetmacro{\xi}{-\dominio}
    \pgfmathsetmacro{\step}{(\dominio-\xi)/28.0}
    \pgfmathsetmacro{\xs}{\xi+\step}
    \foreach \x in {\xi,\xs,...,\dominio}{%
      \pgfmathsetmacro{\radio}{f(\x)}
      \draw[mintDark!70,line width=.48pt,opacity=.32]
        plot[domain=0.55*pi:1.98*pi,smooth,variable=\t]
          ({\radio*cos(\t r)},{\x},{\radio*sin(\t r)});
    }
    \foreach \angulo in {352,336,...,96}{%
      \draw[mintDark!65,line width=.38pt,rotate around y=\angulo,opacity=.22]
        plot[domain=-\dominio:\dominio,smooth,variable=\t] ({0},{\t},{f(\t)});
    }
    \foreach \angulo in {0,16,...,88}{%
      \draw[mintStrong!75,line width=.55pt,rotate around y=\angulo,opacity=.40]
        plot[domain=-\dominio:\dominio,smooth,variable=\t] ({0},{\t},{f(\t)});
    }
    \draw[mintDark!85,line width=.72pt,opacity=.55]
      plot[domain=-\dominio:\dominio,smooth,variable=\t] ({0},{\t},{f(\t)});
  \end{scope}%
}

%% Orbit spheres
\newcommand{\OxfordOrbitSpheres}{%
  \tdplotsetmaincoords{58}{142}%
  \begin{scope}[tdplot_main_coords,scale=2.00,transform shape]
    \def\r{0.14}%
    \def\a{72}%
    \coordinate (OA) at (1,0,0);
    \coordinate (OB) at ({cos(\a)},{sin(\a)},0);
    \draw[mintDark!65,opacity=.42,line width=.45pt] (OA) circle (\r);
    \draw[mintDark!65,opacity=.42,line width=.45pt] (OB) circle (\r);
    \foreach \h in {-\r,0,\r}{%
      \draw[mintDark!55,opacity=.35,line width=.40pt]
        (1,0,\h) arc [radius=1,start angle=0,end angle=\a];
    }
    \begin{scope}[tdplot_screen_coords]
      \fill[mint!80,opacity=.55] (OA) circle (\r);
      \fill[mintDark!65,opacity=.50] (OB) circle (\r);
      \fill[mintDark!95,opacity=.65] (OA) circle (.012);
      \fill[mintDark!95,opacity=.65] (OB) circle (.012);
    \end{scope}
  \end{scope}%
}

%% Hex prism
\newcommand{\OxfordHexPrism}{%
  \tdplotsetmaincoords{66}{38}%
  \begin{scope}[tdplot_main_coords,scale=.86,transform shape,line join=bevel]
    \pgfmathsetmacro{\rhex}{0.82}
    \pgfmathsetmacro{\hhex}{1.55}
    \foreach \P/\ang in {A/0,B/60,C/120,D/180,E/240,F/300}{%
      \coordinate (\P b) at (\ang:\rhex);
      \coordinate (\P t) at ({\rhex*cos(\ang)},{\rhex*sin(\ang)},\hhex);
    }
    \fill[mint!45,opacity=.28] (Ab)--(Bb)--(Bt)--(At)--cycle;
    \fill[mintDark!38,opacity=.20] (Bb)--(Cb)--(Ct)--(Bt)--cycle;
    \draw[mintDark!80,line width=.70pt,opacity=.58]
      (At)--(Bt)--(Ct)--(Dt)--(Et)--(Ft)--cycle;
    \draw[mintDark!60,line width=.52pt,opacity=.38,densely dashed]
      (Ab)--(Bb)--(Cb)--(Db)--(Eb)--(Fb)--cycle;
    \foreach \P in {A,B,C,D,E,F}{%
      \draw[mintDark!65,line width=.52pt,opacity=.45] (\P b)--(\P t);
    }
  \end{scope}%
}

%% Halftone
\newcommand{\OxfordHalftone}{%
  \foreach \i in {0,...,29}{%
    \foreach \j in {0,...,11}{%
      \pgfmathsetmacro{\xx}{1.15 + 0.18*\i + 0.055*\j}
      \pgfmathsetmacro{\yy}{0.40 + 0.105*\j + 0.030*\i}
      \pgfmathsetmacro{\rr}{0.010 + 0.034*exp(-((\i-16)*(\i-16))/130 - ((\j-5)*(\j-5))/18)}
      \fill[mintDark!75,opacity=.38] (\xx,\yy) circle (\rr);
    }%
  }%
}
